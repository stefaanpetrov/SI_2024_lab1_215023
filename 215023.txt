git checkout- za prefrlanje na granka
git checkout -b <new branch> za dodavanje na nov branch
git rebase master- za site granki so master up to date
git push origin master- za spojuvanje na local repositorium
git commit -m "Your commit message"- se pravi commit
git add <file_name>-za staged faza odnosno dodavanje na nov
git clone <repository_URL>-za kloniranje
git merge <branch_name>-za spojuvanje
git status- statusot na grankata koja sakame
git log- za koi logovi se napraveni
